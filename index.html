/** * Cloudflare Worker — Shopify Multi-Store Proxy * Supports Shopify cursor-based pagination (page_info) * Forwards Link header so dashboard can paginate through all records */ const SHOPIFY_API_VERSION = "2025-01"; export default { async fetch(request, env) { const allowedOrigin = env.ALLOWED_ORIGIN || "*"; const url = new URL(request.url); if (request.method === "OPTIONS") { return new Response(null, { headers: corsHeaders(allowedOrigin) }); } // GET /stores if (url.pathname === "/stores") { return await listStores(env, allowedOrigin); } // GET /oauth/{store} if (url.pathname.startsWith("/oauth/")) { const store = url.pathname.split("/")[2]; return await startOAuth(store, env, url, allowedOrigin); } // GET /callback if (url.pathname === "/callback") { return await handleCallback(url, env, allowedOrigin); } // GET /{store}/{endpoint} const parts = url.pathname.replace(/^\//, "").split("/"); if (parts.length >= 2) { const store = parts[0]; const endpoint = parts.slice(1).join("/"); return await proxyRequest(store, endpoint, url.search, env, allowedOrigin); } return jsonError("Invalid route. Usage: /{store}/{endpoint}", 400, allowedOrigin); }, }; // ── LIST STORES ── async function listStores(env, origin) { try { const keys = await env.SHOPIFY_STORES.list(); const stores = new Set(); for (const key of keys.keys) { stores.add(key.name.split(":")[0]); } const storeList = []; for (const store of stores) { const domain = await env.SHOPIFY_STORES.get(`${store}:domain`); const hasToken = !!(await env.SHOPIFY_STORES.get(`${store}:token`)); storeList.push({ id: store, domain, connected: hasToken }); } return new Response(JSON.stringify({ stores: storeList }), { headers: { "Content-Type": "application/json", ...corsHeaders(origin) }, }); } catch (e) { return jsonError("Error listing stores: " + e.message, 500, origin); } } // ── START OAUTH ── async function startOAuth(store, env, url, origin) { const domain = await env.SHOPIFY_STORES.get(`${store}:domain`); const clientId = await env.SHOPIFY_STORES.get(`${store}:client_id`); if (!domain || !clientId) return jsonError(`Store '${store}' not configured`, 404, origin); const redirectUri = `${url.origin}/callback`; const scopes = "read_orders,read_products,read_customers,read_analytics"; const authUrl = `https://${domain}/admin/oauth/authorize?client_id=${clientId}&scope=${scopes}&redirect_uri=${encodeURIComponent(redirectUri)}&state=${store}`; return Response.redirect(authUrl, 302); } // ── OAUTH CALLBACK ── async function handleCallback(url, env, origin) { const code = url.searchParams.get("code"); const state = url.searchParams.get("state"); if (!code || !state) return jsonError("Missing code or state", 400, origin); const clientId = await env.SHOPIFY_STORES.get(`${state}:client_id`); const secret = await env.SHOPIFY_STORES.get(`${state}:secret`); const domain = await env.SHOPIFY_STORES.get(`${state}:domain`); try { const tokenRes = await fetch(`https://${domain}/admin/oauth/access_token`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ client_id: clientId, client_secret: secret, code }), }); const tokenData = await tokenRes.json(); if (!tokenData.access_token) return jsonError("Could not get access token: " + JSON.stringify(tokenData), 500, origin); await env.SHOPIFY_STORES.put(`${state}:token`, tokenData.access_token); return new Response(`

✅ Store ${state} connected successfully!

You can close this window and return to the dashboard.

`, { headers: { "Content-Type": "text/html" } }); } catch (e) { return jsonError("OAuth error: " + e.message, 500, origin); } } // ── PROXY REQUEST ── async function proxyRequest(store, endpoint, queryString, env, origin) { const domain = await env.SHOPIFY_STORES.get(`${store}:domain`); const token = await env.SHOPIFY_STORES.get(`${store}:token`); if (!domain) return jsonError(`Store '${store}' not found`, 404, origin); if (!token) return jsonError(`Store '${store}' not authenticated. Visit /oauth/${store}`, 401, origin); // Parse and clean query params — remove any limit override, force limit=250 const params = new URLSearchParams(queryString); params.set("limit", "250"); // always max const shopifyURL = `https://${domain}/admin/api/${SHOPIFY_API_VERSION}/${endpoint}.json?${params.toString()}`; try { const res = await fetch(shopifyURL, { headers: { "X-Shopify-Access-Token": token, "Content-Type": "application/json", }, }); const data = await res.json(); // Forward Shopify's Link header for cursor pagination const linkHeader = res.headers.get("Link"); const responseHeaders = { "Content-Type": "application/json", ...corsHeaders(origin), }; // Embed link info in response body so CORS doesn't block header access if (linkHeader) { data._link = linkHeader; } return new Response(JSON.stringify(data), { status: res.status, headers: responseHeaders, }); } catch (e) { return jsonError("Error connecting to Shopify: " + e.message, 500, origin); } } // ── HELPERS ── function corsHeaders(origin) { return { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, OPTIONS", "Access-Control-Allow-Headers": "Content-Type", "Access-Control-Expose-Headers": "Link", }; } function jsonError(message, status, origin) { return new Response(JSON.stringify({ error: message }), { status, headers: { "Content-Type": "application/json", ...corsHeaders(origin) }, }); }
